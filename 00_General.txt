	_Navegación Nativa Unfold3D:

		_Pan: mouse left click and drag

		_Orbit: mouse right click and drag

		_Zoom: mouse midle click and drag | mouse scroll up/down

		_A  //zoom a global viewport

		_F  //zoom to selection

		_F1  //point selection

		_F2  //edge selection

		_F3  //faces selection

		_F4  //islands selection


	_Symetry:

		_Alt + S //enable symetry


	_Select Edges:

		_Shift + Mouse Left Click  //select seams

		_Control + mouse Left Click //eliminar selects

		_Space Bar  //borrar todos los edges seleccionados

		_Control + L //selecto loops

		_V: (Path)  //Alt + Drag  /select un edge, y arrastre con "alt", hasta un nuevo destino final del path generado

		_V:  (Loops) //Alt + Drag  /select un edge, y arrastre con "alt", hasta un nuevo destino final del loops generado

		_Control + B  //select borders


	_Create Seams:

		_C  //cut, crear cortes ó seams, a partir de selecciones


	_U  //unfold  //entre mayor sea el número que indiquemos hara un mayor esfuerzo por estirar la maya

UV Panel:
_________


	_Selections:

		_F1  //point selection

		_F2  //edge selection

		_F3  //faces selection

		_F4  //islands selection


	_Q:  //muve grid

	_W:  //translate elementos y sub-elementos (la herramienta es multi funcional)

	_E: //rotar elementos y sub-elementos (la herramienta es multi funcional)

	_R:  //scale elementos y sub-elementos (la herramienta es multi funcional)


	_O:  //optimized island, ó cualquier seleccion ó todo ni nada está seleccionado

	
	_Weld/Soldado:

		_Control + Shift + C  //soldar dos ó más sub-elementos seleccionados



	Align/Alineamiento:

		_Parallelize Point: Control + Shift A  //alinea puntos seleccionados en el ángulo más cercano

		_Horizontaleze Point: Control + Shift H  //alinea puntos seleccionados en el ángulo más cercano

		_Verticalize Point: Control + Shift V  //alinea puntos seleccionados en el ángulo más cercano

		_Edges Horizontal: Control + Alt + H //alinear edges horizontalmente para cuando hagamos "optimized", "K"

		_Edges Vertical: Control + Alt + V //alinear edges horizontalmente para cuando hagamos "optimized", "K"



	_Pin/Sostener:

		_Pin: Alt + P  //seleccionar sub-elementos para que éstos queden fijos en la optimización de proyecciones
